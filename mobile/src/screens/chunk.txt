
                                <TouchableOpacity style={styles.rescanBtn} onPress={handleScanPress}>
                                    <Camera size={20} color="#FFF" />
                                    <Text style={styles.rescanText}>SCAN ANOTHER ITEM</Text>
                                </TouchableOpacity>
                            </View>
                        ) : (
                            <View style={styles.emptyScanContainer}>
                                <Text style={styles.emptyScanText}>Press the button below to start scanning</Text>
                            </View>
                        )}
                    </View>
                </View>
            )}
        </View>
    );
};
